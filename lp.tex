\subsection{Conjunção}

Introdução do "e":
 \begin{prooftree}
     \AxiomC{A}
     \AxiomC{B}
     \BinaryInfC{$A \land B$}
\end{prooftree}


Exclusão do "e":
 \begin{prooftree}
     \AxiomC{$A \land B$}
     \UnaryInfC{A}
\end{prooftree}

\subsection{Disjunção}

Introdução do "ou":
\begin{prooftree}
     \AxiomC{A}
     \UnaryInfC{$A \lor B$}
\end{prooftree}

Exclusão do "ou": 

  \begin{prooftree}
        \AxiomC{$A \lor B$}
              \AxiomC{$A$}
              \UnaryInfC{$C$}
                   \AxiomC{$B$}
                   \UnaryInfC{$C$}
              \TrinaryInfC{$C$}
     \end{prooftree}
